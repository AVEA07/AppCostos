CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    //apellido
    usuario VARCHAR(50) UNIQUE NOT NULL,
    contrasena VARCHAR(255) NOT NULL,
    rango_id INT,
    correo VARCHAR(100),
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (rango_id) REFERENCES rangos(id)
); 

CREATE TABLE rangos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_rango VARCHAR(50) NOT NULL,
    nivel INT NOT NULL
); 

CREATE TABLE proyectos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    usuario_id INT,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
); 

CREATE TABLE costos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    proyecto_id INT,
    nombre_modulo VARCHAR(100),
    complejidad ENUM('baja', 'media', 'alta'),
    horas_estimadas DECIMAL(10,2),
    costo_por_hora DECIMAL(10,2),
    costo_total DECIMAL(10,2) GENERATED ALWAYS AS (horas_estimadas * costo_por_hora) STORED,
    FOREIGN KEY (proyecto_id) REFERENCES proyectos(id)
); 

CREATE TABLE historial (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT,
    accion VARCHAR(255),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);
